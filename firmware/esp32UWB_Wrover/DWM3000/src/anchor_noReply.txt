#include <Arduino.h>
#include <DW3000.h>

const uint8_t PIN_RST = 27;
const uint8_t PIN_IRQ = 34;
const uint8_t PIN_SS = 4;

static dwt_config_t config = {
  5, DWT_PLEN_128, DWT_PAC8, 9, 9, 1, DWT_BR_6M8, DWT_PHRMODE_STD, DWT_PHRRATE_STD, 
  (129 + 8 - 8), DWT_STS_MODE_OFF, DWT_STS_LEN_64, DWT_PDOA_M0
};

#define TX_ANT_DLY 16385
#define RX_ANT_DLY 16385

static uint8_t tx_poll_msg[] = {0x41, 0x88, 0, 0xCA, 0xDE, 'P', 'O', 'L', '1', 0x21, 0, 0};
static uint8_t frame_seq_nb = 0;

extern dwt_txconfig_t txconfig_options;

void setup() {
    Serial.begin(115200);
    delay(2000);
    Serial.println("==================================");
    Serial.println(" KOTWICA-GADUŁA START ");
    Serial.println("==================================");

    spiBegin(PIN_IRQ, PIN_RST);
    spiSelect(PIN_SS);
    delay(2);

    while (!dwt_checkidlerc()) { Serial.println("IDLE FAILED"); delay(1000); }
    if (dwt_initialise(DWT_DW_INIT) == DWT_ERROR) { Serial.println("INIT FAILED"); while (1); }
    
    dwt_setleds(DWT_LEDS_ENABLE | DWT_LEDS_INIT_BLINK);
    if (dwt_configure(&config)) { Serial.println("CONFIG FAILED"); while (1); }
    dwt_configuretxrf(&txconfig_options);
    dwt_setrxantennadelay(RX_ANT_DLY);
    dwt_settxantennadelay(TX_ANT_DLY);
    
    Serial.println("Gotowe. Zaczynam nadawać w ciemno!");
}

void loop() {
    tx_poll_msg[2] = frame_seq_nb; // Aktualizacja numeru wiadomości
    
    dwt_write32bitreg(SYS_STATUS_ID, SYS_STATUS_TXFRS_BIT_MASK);
    dwt_writetxdata(sizeof(tx_poll_msg), tx_poll_msg, 0);
    dwt_writetxfctrl(sizeof(tx_poll_msg), 0, 1);

    // KRYTYCZNA ZMIANA: Wysyłamy natychmiast i NIE CZEKAMY na odpowiedź
    int ret = dwt_starttx(DWT_START_TX_IMMEDIATE);

    if (ret == DWT_SUCCESS) {
        // Czekamy ułamek sekundy, aż fala fizycznie opuści antenę
        while (!(dwt_read32bitreg(SYS_STATUS_ID) & SYS_STATUS_TXFRS_BIT_MASK)) {};
        dwt_write32bitreg(SYS_STATUS_ID, SYS_STATUS_TXFRS_BIT_MASK);
        Serial.println(">>> Wysłano impuls radiowy (POLL)! <<<");
    } else {
        Serial.println("[BŁĄD] Chip DW3000 odmówił nadania pakietu!");
    }

    delay(250); // Nadajemy 4 razy na sekundę
    frame_seq_nb++;
}
